then runs DPO fine-tuning. This gives the model a stronger quality signal than the
hand-crafted responses in the original dataset.

- `data/fluento_dpo_500.jsonl` — original DPO dataset
- `data/gemini_pro_responses.jsonl` — generated by `gen_gemini_responses.py`

## Cell 3: Build the merged training dataset

Merges the two files: Gemini Pro response → `chosen`, original `rejected` → `rejected`.

```python
import json

ORIGINAL_FILE = "/content/drive/MyDrive/fluento_dpo_500.jsonl"
GEMINI_FILE   = "/content/drive/MyDrive/gemini_pro_responses.jsonl"

# Load original pairs
original = {}
with open(ORIGINAL_FILE) as f:
    for line in f:
        row = json.loads(line)
        original[row["id"]] = row

# Load Gemini Pro responses
gemini = {}
with open(GEMINI_FILE) as f:
    for line in f:
        row = json.loads(line)
        gemini[row["id"]] = row

# Merge — only include IDs present in both files
merged = []
skipped = 0
for id_, orig in original.items():
    if id_ not in gemini:
        skipped += 1
        continue
    merged.append({
        "id":            id_,
        "dimension":     orig["dimension"],
        "learner_level": orig["learner_level"],
        "prompt":        orig["prompt"],
        "chosen":        gemini[id_]["gemini_pro_response"],
        "rejected":      orig["rejected"],
    })

print(f"Merged {len(merged)} pairs  ({skipped} skipped — Gemini response missing)")
print(f"\nSample:")
print(f"  PROMPT:   {merged[0]['prompt']}")
print(f"  CHOSEN:   {merged[0]['chosen']}")
print(f"  REJECTED: {merged[0]['rejected']}")
```

## Cell 5: Load base model

```python
from transformers import AutoModelForCausalLM, AutoTokenizer, BitsAndBytesConfig
import torch

model_name = "Qwen/Qwen2.5-7B-Instruct"

bnb_config = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_quant_type="nf4",
    bnb_4bit_compute_dtype=torch.float16,
    bnb_4bit_use_double_quant=True,
)

tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForCausalLM.from_pretrained(
    model_name,
    quantization_config=bnb_config,
    device_map="auto",
    torch_dtype=torch.float16,
)

print(f"Loaded {model_name}")
```

## Cell 6: Add LoRA adapters

```python
from peft import LoraConfig, get_peft_model, prepare_model_for_kbit_training

model = prepare_model_for_kbit_training(model)

lora_config = LoraConfig(
    r=16,
    lora_alpha=16,
    target_modules=["q_proj", "k_proj", "v_proj", "o_proj",
                    "gate_proj", "up_proj", "down_proj"],
    lora_dropout=0.05,
    bias="none",
    task_type="CAUSAL_LM",
)
model = get_peft_model(model, lora_config)
```

## Cell 7: Prepare dataset

```python
from datasets import load_dataset

dataset = load_dataset("json", data_files=MERGED_FILE, split="train")

SYSTEM = (
    "You are a voice English language tutor. "
    "Keep responses to 1-2 sentences. "
    "Ask guiding questions. Match the learner's level."
)

def format_for_dpo(example):
    return {
        "prompt": [
            {"role": "system", "content": SYSTEM},
            {"role": "user",   "content": example["prompt"]},
        ],
        "chosen":   [{"role": "assistant", "content": example["chosen"]}],
        "rejected": [{"role": "assistant", "content": example["rejected"]}],
    }

dpo_dataset = dataset.map(format_for_dpo)
print(f"Dataset ready: {len(dpo_dataset)} pairs")
```

---

## Cell 8: Train

```python
from trl import DPOConfig, DPOTrainer

training_args = DPOConfig(
    output_dir="./tutor-gemini-dpo-output",
    per_device_train_batch_size=2,
    gradient_accumulation_steps=4,
    num_train_epochs=2,
    learning_rate=5e-6,
    beta=0.1,
    max_length=1024,
    max_prompt_length=512,
    logging_steps=10,
    save_steps=100,
    warmup_ratio=0.1,
    optim="adamw_8bit",
    fp16=not torch.cuda.is_bf16_supported(),
    bf16=torch.cuda.is_bf16_supported(),
    seed=42,
)

trainer = DPOTrainer(
    model=model,
    args=training_args,
    train_dataset=dpo_dataset,
    processing_class=tokenizer,
)

print("Starting DPO fine-tuning with Gemini Pro responses...")
trainer.train()
print("Done!")
```

## Cell 10: Compare — base vs v1 vs Gemini-trained

Quick side-by-side to see if the new model improved:

```python
from peft import PeftModel

# Load v1 (original fine-tune) for comparison
V1_PATH = "/content/drive/MyDrive/tutor-dpo-lora"
base_for_v1 = AutoModelForCausalLM.from_pretrained(
    model_name, quantization_config=bnb_config, device_map="auto", torch_dtype=torch.float16
)
model_v1 = PeftModel.from_pretrained(base_for_v1, V1_PATH)
model_v1.eval()

model.eval()

def ask(m, prompt, max_tokens=150):
    messages = [{"role": "system", "content": SYSTEM}, {"role": "user", "content": prompt}]
    inputs = tokenizer.apply_chat_template(
        messages, tokenize=True, add_generation_prompt=True,
        return_tensors="pt", return_dict=True
    ).to("cuda")
    outputs = m.generate(**inputs, max_new_tokens=max_tokens, temperature=0.7, top_p=0.9, do_sample=True)
    return tokenizer.decode(outputs[0][inputs["input_ids"].shape[-1]:], skip_special_tokens=True).strip()

test_prompts = [
    ("beginner",     "I goed to the store yesterday."),
    ("beginner",     "What is the past tense of 'go'?"),
    ("intermediate", "She said me that she was coming."),
    ("advanced",     "Between you and I, she's going to resign."),
]

for level, prompt in test_prompts:
    print(f"\n{'='*70}")
    print(f"[{level.upper()}] {prompt}")
    print(f"  v1 (hand-crafted):  {ask(model_v1, prompt)}")
    print(f"  v2 (Gemini Pro):    {ask(model, prompt)}")
```
